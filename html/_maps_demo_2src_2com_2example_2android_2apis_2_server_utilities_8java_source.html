33  Log.i(TAG, "registering device (regId = " + regId + ")");
34  String serverUrl = SERVER_URL;
35  Map<String, String> params = new HashMap<String, String>();
36  params.put("regId", regId);
37  params.put("name", name);
38  params.put("email", email);
39 
40  long backoff = BACKOFF_MILLI_SECONDS + random.nextInt(1000);
41  // Once GCM returns a registration id, we need to register on our server
42  // As the server might be down, we will retry it a couple
43  // times.
44  for (int i = 1; i <= MAX_ATTEMPTS; i++) {
45  Log.d(TAG, "Attempt #" + i + " to register");
46  try {
47  displayMessage(context, context.getString(
48  R.string.server_registering, i, MAX_ATTEMPTS));
49  post(serverUrl, params);
50  GCMRegistrar.setRegisteredOnServer(context, true);
51  String message = context.getString(R.string.server_registered);
52  CommonUtilities.displayMessage(context, message);
53  return;
54  } catch (IOException e) {
55  // Here we are simplifying and retrying on any error; in a real
56  // application, it should retry only on unrecoverable errors
57  // (like HTTP error code 503).
58  Log.e(TAG, "Failed to register on attempt " + i + ":" + e);
59  if (i == MAX_ATTEMPTS) {
60  break;
61  }
62  try {
63  Log.d(TAG, "Sleeping for " + backoff + " ms before retry");
64  Thread.sleep(backoff);
65  } catch (InterruptedException e1) {
66  // Activity finished before we complete - exit.
67  Log.d(TAG, "Thread interrupted: abort remaining retries!");
68  Thread.currentThread().interrupt();
69  return;
70  }
71  // increase backoff exponentially
72  backoff *= 2;
73  }
74  }
75  String message = context.getString(R.string.server_register_error,
76  MAX_ATTEMPTS);
77  CommonUtilities.displayMessage(context, message);
78  }
